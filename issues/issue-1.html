What is a Kolmogorov-Arnold Network?

KANs replace fixed activation functions with learnable splines.

# Just for show – full repo is separate
class DDN(nn.Module):
    def __init__(self, input_dim=8, hidden_layers=4, neurons = 128, output_dim=1):
        
        super().__init__()

        layers = []

        # Input layer:
        layers.append(nn.Linear(input_dim, neurons))
        layers.append(nn.SiLU())


        for _ in range(hidden_layers-1): # OBS input är "hidden", hence "-1"
            layers.append(nn.Linear(neurons, neurons))
            layers.append(nn.SiLU())
        

        self.network = nn.Sequential(*layers)

        # Output layer
        self.output_price = nn.Linear(neurons, output_dim)
        self.softplus = nn.Softplus() # Garanterar positiva priser

        self._init_weights()

    def forward(self, x):
        x1 = self.network(x)
        y = self.output_price(x1)
        return self.softplus(y) # > 0 !
